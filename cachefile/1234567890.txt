	function fnInitDates()
	{
		var Datum = new Date()	
		Datum = new Date(Datum.getFullYear(),Datum.getMonth(),Datum.getDate(), 0, 0, 0, 0)	

		
		var EindDatum = new Date(Datum.getTime() + (8*24*60*60*1000) - 1)
		// initialize startdate
		if (CurrentFilter.fnGetFilterItem('di') == null)
			document.getElementById("tbxTournamentStartDate").value = Datum.getDate() + "-"  + (1+Datum.getMonth()) + "-" + Datum.getFullYear()
		else
		{
			document.getElementById("tbxTournamentStartDate").value = CurrentFilter.fnGetFilterItem('di').fnGetFilterValue(FilterType.Date)
		}

	//initialize enddate
		if (CurrentFilter.fnGetFilterItem('da') == null)
			document.getElementById("tbxTournamentEndDate").value = EindDatum.getDate() + "-"  + (1+EindDatum.getMonth()) + "-" + EindDatum.getFullYear()
		else
		{
			document.getElementById("tbxTournamentEndDate").value = CurrentFilter.fnGetFilterItem('da').fnGetFilterValue(FilterType.Date)
		}	
	}
	
	if (document.getElementById("tbxTournamentStartDate"))
	{
		fnInitDates()
	}

	{
		//newFilter = new DocumentFilter(CurrentFilter.strBaseUrl)

		CurrentFilter.fnSetFilterItem('da', FilterType.Date, document.getElementById("tbxTournamentEndDate").value)

		CurrentFilter.fnExecuteFilter()
	}
